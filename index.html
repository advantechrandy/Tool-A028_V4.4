// 核心計算邏輯 function calculate() { const L = parseFloat(sizeX.value); const W = parseFloat(sizeY.value); const alpha = parseFloat(cte.value) * 1e-6; const E = parseFloat(youngs.value); const dT = parseFloat(dT_input.value); const K = parseFloat(kSlider.value); const nu = 0.15; // Poisson's Ratio // 1. Thermal Strain: ε = α * ΔT const strain = alpha * dT; // 2. Biaxial Stress: σ = K * (E * ε) / (1 - ν) const stress = K * (E * strain) / (1 - nu); // 3. Linear Deformation: ΔL = L * ε * (1 - K) const dL = L * strain * (1 - K); const dW = W * strain * (1 - K); updateUI(stress, dL, dW, strain); }
